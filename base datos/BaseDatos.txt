Create table TipoUsuarios(
id int(11) NOT NULL PRIMARY KEY AUTO_INCREMENT,
title varchar(100) NOT NULL,
description varchar(100) not null,
published BOOLEAN DEFAULT false
);

Create table Ubicaciones(
id int(11) NOT NULL PRIMARY KEY AUTO_INCREMENT,
title varchar(100) NOT NULL,
description varchar(100) not null,
published BOOLEAN DEFAULT false
);

Create table TipoReporteContenedores(
Id_TipoReporteContenedor char(8) primary key,
DescTipoReporteContenedor varchar(20) not null
);

Create table TipoReporteMateriales(
Id_TipoReporteMaterial char(8) primary key,
DescTipoReporteMaterial varchar(20) not null
);

Create table TipoContenedores(
Id_TipoContenedor char(8) primary key,
DescTipoContenedor varchar(20) not null
);

Create table TipoMateriales(
Id_TipoMaterial char(8) primary key,
DescTipoMaterial varchar(20) not null
);



Create table Areas(
Id_Area char(8) primary key,
DescArea varchar(20) not null
);

Create table Arboles(
Id_Arbol char(8) primary key,
DescArbol varchar(20) not null
);

Create table TipoHerramientas(
Id_TipoHerramienta char(8) primary key,
DescTipoHerramienta varchar(20) not null
);



Create table Usuarios(
Id_Usuario char(8) primary key,
Id_TipoUsuario int(11) not null AUTO_INCREMENT,
NombreUsuario varchar(20) not null,
ApellidoP varchar(20) not null,
ApellidoM varchar(20) not null,
Password char(8),
CONSTRAINT fk_TipoUsuario FOREIGN KEY (Id_TipoUsuario) REFERENCES TipoUsuarios (id)
);



Create table ModuloContenedores(
Id_ModuloContenedor char(8) primary key,
Id_Ubicacion int(11) not null AUTO_INCREMENT,
DescModuloContenedor varchar(20) not null,
CONSTRAINT fk_Ubicacion FOREIGN KEY (Id_Ubicacion) REFERENCES Ubicaciones (id)
);

Create table ModuloMaterial(
Id_ModuloMaterial char(8) primary key,
Id_Area char(8) not null,
DescModuloMaterial varchar(20) not null,
CONSTRAINT fk_Area FOREIGN KEY (Id_Area) REFERENCES Areas (Id_Area)
);


Create table Materiales(
Id_Material char(8) primary key,
Id_TipoMaterial char(8) not null,
Id_ModuloMaterial char(8) not null,
DescMaterial varchar(20) not null,
CONSTRAINT fk_TipoMaterial FOREIGN KEY (Id_TipoMaterial) REFERENCES TipoMateriales (Id_TipoMaterial),
CONSTRAINT fk_ModuloMaterial FOREIGN KEY (Id_ModuloMaterial) REFERENCES ModuloMaterial (Id_ModuloMaterial)
);

Create table Herramientas(
Id_Herramienta char(8) primary key,
Id_TipoHerramienta char(8) not null,
Id_ModuloMaterial2 char(8) not null,
DescMaterial varchar(20) not null,
CONSTRAINT fk_TipoHerramienta FOREIGN KEY (Id_TipoHerramienta) REFERENCES TipoHerramientas (Id_TipoHerramienta),
CONSTRAINT fk_ModuloMaterial2 FOREIGN KEY (Id_ModuloMaterial2) REFERENCES ModuloMaterial (Id_ModuloMaterial)
);

Create table Contenedores(
Id_Contenedor char(8) primary key,
Id_TipoContenedor char(8) not null,
Id_ModuloContenedor char(8) not null,
DescContenedor varchar(20) not null,
CONSTRAINT fk_TipoContenedor FOREIGN KEY (Id_TipoContenedor) REFERENCES TipoContenedores (Id_TipoContenedor),
CONSTRAINT fk_ModuloContenedor FOREIGN KEY (Id_ModuloContenedor) REFERENCES ModuloContenedores (Id_ModuloContenedor)
);

Create table ReportesArboles(
Id_ReporteArbol char(8) primary key,
Id_Usuario char(8) not null,
Id_Arbol char(8) not null,
DescReporteArbol varchar(20) not null,
CONSTRAINT fk_Arbol FOREIGN KEY (Id_Arbol) REFERENCES Arboles (Id_Arbol),
CONSTRAINT fk_Usuario FOREIGN KEY (Id_Usuario) REFERENCES Usuarios (Id_Usuario)
);

Create table ReportesContenedores(
Id_ReporteContenedor char(8) primary key,
Id_Usuario1 char(8) not null,
Id_ModuloContenedor2 char(8) not null,
Id_TipoReporteContenedor char(8) not null,
DescReporteContenedor varchar(20) not null,
CONSTRAINT fk_TipoReporteContenedor FOREIGN KEY (Id_TipoReporteContenedor) REFERENCES TipoReporteContenedores (Id_TipoReporteContenedor),
CONSTRAINT fk_ModuloContenedor2 FOREIGN KEY (Id_ModuloContenedor2) REFERENCES ModuloContenedores (Id_ModuloContenedor),
CONSTRAINT fk_Usuario1 FOREIGN KEY (Id_Usuario1) REFERENCES Usuarios (Id_Usuario)
);

Create table ReportesMateriales(
Id_ReporteMaterial char(8) primary key,
Id_Usuario2 char(8) not null,
Id_ModuloMaterial3 char(8) not null,
Id_TipoReporteMaterial char(8) not null,
DescReporteMaterial varchar(20) not null,
CONSTRAINT fk_TipoReporteMaterial FOREIGN KEY (Id_TipoReporteMaterial) REFERENCES TipoReporteMateriales (Id_TipoReporteMaterial),
CONSTRAINT fk_ModuloMaterial3 FOREIGN KEY (Id_ModuloMaterial3) REFERENCES ModuloMaterial (Id_ModuloMaterial),
CONSTRAINT fk_Usuario2 FOREIGN KEY (Id_Usuario2) REFERENCES Usuarios (Id_Usuario)
);
